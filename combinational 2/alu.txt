// alu
module alu(input  a,b,
           input [3:0] operation ,
          output reg y
          );
  
  always @ (*) 
    begin

        case (operation)
    4'b0000: y = a+b;
    4'b0001: y = a-b;
    4'b0010: y = a^b;
    4'b0011: y = a/b;
    4'b0100: y = a&&b;
    4'b0101: y =a |b;
    4'b0110: y = a&b;
    default: y =~a;
  endcase
    endmodule
